Google Sheets Integration

Setup (one time only)

1. Open Google Sheets and create a new sheet named Gym Log

2. Go to Extensions → Apps Script

3. Delete any existing code and paste in the script below

4. Click Deploy → New Deployment → Web App

5. Set Execute as: Me  |  Who has access: Anyone

6. Copy the Web App URL and paste it below

Apps Script Code

function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName('Gym Log'); if (!sheet) { sheet = SpreadsheetApp.getActiveSpreadsheet() .insertSheet('Gym Log'); sheet.appendRow(['Date','Exercise','S1 Wt','S1 Reps', 'S2 Wt','S2 Reps','S3 Wt','S3 Reps', 'S4 Wt','S4 Reps','Notes','Muscle','Est Cal']); } var rows = JSON.parse(e.postData.contents); rows.forEach(function(r){ sheet.appendRow(r); }); return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'alive'})) .setMimeType(ContentService.MimeType.JSON); }
